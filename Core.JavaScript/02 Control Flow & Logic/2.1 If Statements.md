# **[[2.1 If Statements]]**

## Overview

This topic covers conditional logic using if statements. You'll learn the if condition syntax, if-else branching, else if chains for multiple conditions, nested if structures, conditional logic patterns, comparison operators in conditions, and boolean expressions for complex decisions.

## If Condition

The `if` statement executes code when a condition is true.

**Basic Syntax:**
```javascript
if (condition) {
    // Code executes when condition is true
}
```

**Simple Examples:**
```javascript
let age = 18;

if (age >= 18) {
    console.log("You are an adult");
}

let isLoggedIn = true;

if (isLoggedIn) {
    console.log("Welcome back!");
}

let temperature = 35;

if (temperature > 30) {
    console.log("It's hot outside");
}
```

**Truthy Conditions:**
```javascript
let username = "John";

if (username) {
    console.log("Username exists:", username);
}

let items = [1, 2, 3];

if (items.length) {
    console.log("Array has items");
}
```

## If-Else

The `else` clause provides an alternative when the condition is false.

**Syntax:**
```javascript
if (condition) {
    // Executes when condition is true
} else {
    // Executes when condition is false
}
```

**Examples:**
```javascript
let score = 65;

if (score >= 60) {
    console.log("You passed!");
} else {
    console.log("You failed. Try again.");
}

let hour = 14;

if (hour < 12) {
    console.log("Good morning");
} else {
    console.log("Good afternoon");
}

let user = null;

if (user) {
    console.log("User:", user.name);
} else {
    console.log("No user logged in");
}
```

## Else If

Use `else if` to check multiple conditions in sequence.

**Syntax:**
```javascript
if (condition1) {
    // Executes when condition1 is true
} else if (condition2) {
    // Executes when condition2 is true
} else if (condition3) {
    // Executes when condition3 is true
} else {
    // Executes when no condition is true
}
```

**Grade Example:**
```javascript
let score = 85;
let grade;

if (score >= 90) {
    grade = "A";
} else if (score >= 80) {
    grade = "B";
} else if (score >= 70) {
    grade = "C";
} else if (score >= 60) {
    grade = "D";
} else {
    grade = "F";
}

console.log("Your grade:", grade);  // "B"
```

**Time-Based Greeting:**
```javascript
let hour = new Date().getHours();
let greeting;

if (hour < 6) {
    greeting = "Good night";
} else if (hour < 12) {
    greeting = "Good morning";
} else if (hour < 18) {
    greeting = "Good afternoon";
} else if (hour < 22) {
    greeting = "Good evening";
} else {
    greeting = "Good night";
}

console.log(greeting);
```

**Important:** Only the first matching condition executes:
```javascript
let value = 15;

if (value > 5) {
    console.log("Greater than 5");  // This runs
} else if (value > 10) {
    console.log("Greater than 10");  // Skipped (first match won)
}
```

## Nested If

Place if statements inside other if statements for complex logic.

**Syntax:**
```javascript
if (outerCondition) {
    if (innerCondition) {
        // Both conditions are true
    }
}
```

**Login Validation Example:**
```javascript
let username = "admin";
let password = "secret123";
let isActive = true;

if (username === "admin") {
    if (password === "secret123") {
        if (isActive) {
            console.log("Login successful!");
        } else {
            console.log("Account is disabled");
        }
    } else {
        console.log("Wrong password");
    }
} else {
    console.log("User not found");
}
```

**Refactored with Early Returns:**
```javascript
function login(username, password, isActive) {
    if (username !== "admin") {
        return "User not found";
    }
    
    if (password !== "secret123") {
        return "Wrong password";
    }
    
    if (!isActive) {
        return "Account is disabled";
    }
    
    return "Login successful!";
}
```

**Nested with Else:**
```javascript
let age = 25;
let hasLicense = true;

if (age >= 18) {
    if (hasLicense) {
        console.log("You can drive");
    } else {
        console.log("Get a license first");
    }
} else {
    console.log("Too young to drive");
}
```

## Conditional Logic

Build effective conditional logic using patterns and best practices.

**Single Statement (Braces Optional):**
```javascript
// Without braces (single statement only)
if (isActive) console.log("Active");

// With braces (recommended for clarity)
if (isActive) {
    console.log("Active");
}
```

**Combining Conditions:**
```javascript
let age = 25;
let hasTicket = true;

// AND: Both must be true
if (age >= 18 && hasTicket) {
    console.log("You can enter");
}

// OR: At least one must be true
if (age < 12 || age >= 65) {
    console.log("You get a discount");
}

// NOT: Inverts the condition
if (!isLoggedIn) {
    console.log("Please log in");
}
```

**Guard Clauses:**
```javascript
function processOrder(order) {
    // Guard clauses handle invalid cases first
    if (!order) {
        console.log("No order provided");
        return;
    }
    
    if (order.items.length === 0) {
        console.log("Order is empty");
        return;
    }
    
    if (!order.customer) {
        console.log("No customer info");
        return;
    }
    
    // Main logic (only valid orders reach here)
    console.log("Processing order...");
}
```

## Comparison Operators

Operators used to build conditions.

**Equality:**
```javascript
let x = 5;

// Strict equality (recommended)
if (x === 5) { }  // true
if (x === "5") { }  // false (different types)

// Loose equality (avoid)
if (x == "5") { }  // true (type coercion)

// Strict inequality
if (x !== 10) { }  // true
```

**Relational:**
```javascript
let age = 25;

if (age < 18) { }   // Less than
if (age > 18) { }   // Greater than
if (age <= 25) { }  // Less than or equal
if (age >= 21) { }  // Greater than or equal
```

**String Comparison:**
```javascript
// Lexicographic (dictionary) order
if ("apple" < "banana") { }  // true
if ("Apple" < "apple") { }   // true (uppercase first)

// Case-insensitive comparison
let input = "Hello";
if (input.toLowerCase() === "hello") {
    console.log("Match!");
}
```

## Boolean Expressions

Combine multiple conditions into complex expressions.

**Logical AND (&&):**
```javascript
let age = 25;
let hasID = true;
let isMember = true;

// All conditions must be true
if (age >= 21 && hasID && isMember) {
    console.log("Full access granted");
}
```

**Logical OR (||):**
```javascript
let role = "admin";

// At least one condition must be true
if (role === "admin" || role === "moderator") {
    console.log("Access to dashboard");
}
```

**Logical NOT (!):**
```javascript
let isBlocked = false;

if (!isBlocked) {
    console.log("User can post");
}
```

**Complex Expressions:**
```javascript
let user = {
    age: 25,
    isPremium: true,
    country: "USA"
};

// Parentheses for grouping
if ((user.age >= 18 && user.country === "USA") || user.isPremium) {
    console.log("Content available");
}

// Equivalent with variables for clarity
let isAdultInUSA = user.age >= 18 && user.country === "USA";
let hasPremiumAccess = user.isPremium;

if (isAdultInUSA || hasPremiumAccess) {
    console.log("Content available");
}
```

**Short-Circuit Evaluation:**
```javascript
// && stops at first false
let result = true && false && expensiveOperation();  // expensiveOperation not called

// || stops at first true
let value = null || "default";  // "default"

// Practical usage
let user = getUser();
let name = user && user.name;  // Safe property access
```

## Summary / Key Takeaways

- **if statement:** Executes code when condition evaluates to true
- **if-else:** Provides alternative path when condition is false
- **else if:** Chain multiple conditions; first match wins
- **Nested if:** Complex logic with inner conditions; consider refactoring with early returns
- **Comparison operators:** Use `===` and `!==` (strict equality); avoid `==` and `!=`
- **Boolean expressions:** Combine with `&&` (AND), `||` (OR), `!` (NOT)
- **Guard clauses:** Handle invalid cases early for cleaner code
- **Best practices:** Use braces consistently, prefer early returns, extract complex conditions to variables

If statements are the foundation of decision-making in JavaScript programs.
